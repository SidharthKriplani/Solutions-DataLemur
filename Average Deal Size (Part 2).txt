WITH joined_data as 
(
SELECT 
co.customer_id AS customer_id, 
co.num_seats AS num_seats, 
co.yearly_seat_cost AS yearly_seat_cost, 
cu.name AS name, 
cu.employee_count AS employee_count

FROM contracts co
 
JOIN customers cu 
ON co.customer_id = cu.customer_id
), 

smb AS
(
SELECT 
SUM(yearly_seat_cost*num_seats)
/
COUNT(DISTINCT name)

FROM joined_data
WHERE employee_count < 100
), 

midmarket AS
(
SELECT 
SUM(yearly_seat_cost*num_seats)
/
COUNT(DISTINCT name)

FROM joined_data
WHERE 
employee_count >=100 
AND 
employee_count <=999
), 

enterprise AS 
(
SELECT 
SUM(yearly_seat_cost*num_seats)
/
COUNT(DISTINCT name)
FROM joined_data
WHERE employee_count>=1000
)


SELECT *, 
(SELECT * FROM midmarket), 
(SELECT * FROM enterprise)
FROM smb
